    suspend fun `navigate to child by index`() {

        mockServer.`Given server for url answers`(
                url,
                """
                {
                    "accounts": [
                        {
                            "name": "John",
                            "amount": 100
                        },
                        {
                            "name": "Jane",
                            "amount": 200
                        }
                    ]
                }
                """)

        request {
            baseUri(mockServer.baseUrl())
            get(url)
        }

        bodyJson()["account"][0]["name"].isEquals("Jane")
